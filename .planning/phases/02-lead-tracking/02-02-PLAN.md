D. **Replace `renderLeads` function entirely** (lines 778-810):
```javascript
function renderLeads(leads) {
  const tbody = document.getElementById('leadsTableBody');
  const suffix = isDemo ? ' (sample data)' : '';
  document.getElementById('leadsCount').textContent =
    leads.length + ' lead' + (leads.length !== 1 ? 's' : '') + suffix;

  if (!leads.length) {
    tbody.innerHTML = '<tr><td colspan="5" class="leads-empty">No leads captured yet. When a bot detects a lead, it will appear here.</td></tr>';
    return;
  }

  tbody.innerHTML = leads.map(l => {
    const name = escHtml(l.name || '\u2014');
    const biz  = escHtml(l.business || '\u2014');
    const bot  = escHtml(l.bot_name || '\u2014');
    const msg  = escHtml(l.message || '');
    const date = l.created_at
      ? new Date(l.created_at).toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' })
      : '\u2014';
    const time = l.created_at
      ? new Date(l.created_at).toLocaleTimeString('en-US', { hour: 'numeric', minute: '2-digit' })
      : '';
    const statusClass = {
      'New':       'status-new',
      'Contacted': 'status-contacted',
      'Closed':    'status-closed',
      'Lost':      'status-lost',
    }[l.status] || 'status-new';

    const selectHtml = `<select class="lead-status-select ${statusClass}"
      onchange="updateLeadStatus('${escHtml(l.id)}', this.value); this.className='lead-status-select status-'+this.value.toLowerCase();">
      ${['New','Contacted','Closed','Lost'].map(s =>
        '<option value="' + s + '"' + (l.status === s ? ' selected' : '') + '>' + s + '</option>'
      ).join('')}
    </select>`;

    return '<tr>' +
      '<td><div class="lead-name">' + name + '</div>' +
        (msg ? '<div class="lead-msg" title="' + msg + '">' + msg + '</div>' : '') +
      '</td>' +
      '<td>' + biz + '</td>' +
      '<td>' + bot + '</td>' +
      '<td>' + selectHtml + '</td>' +
      '<td style="white-space:nowrap; color:var(--muted); font-size:0.82rem;">' + date + '<br><span style="font-size:0.72rem;">' + time + '</span></td>' +
    '</tr>';
  }).join('');
}
```

E. **Add `updateLeadStatus` function** immediately after `renderLeads`:
```javascript
async function updateLeadStatus(leadId, newStatus) {
  if (isDemo) return;
  const { error } = await sb
    .from('leads')
    .update({ status: newStatus })
    .eq('id', leadId);
  if (error) console.warn('Status update failed (non-fatal):', error.message);
}
```

F. **Replace `exportLeads` function** (lines 812-828) to use new schema:
```javascript
async function exportLeads() {
  let data = isDemo ? DEMO_LEADS : null;
  if (!data) {
    const { data: d } = await sb.from('leads').select('*').order('created_at', { ascending: false });
    data = d || [];
  }
  if (!data.length) { alert('No leads to export yet.'); return; }
  const csv = ['Date,Name,Business,Bot,Status,Message']
    .concat(data.map(r =>
      '"' + (r.created_at||'') + '","' + (r.name||'') + '","' + (r.business||'') + '","' + (r.bot_name||'') + '","' + (r.status||'') + '","' + ((r.message||'').replace(/"/g,'""')) + '"'
    )).join('\n');
  const blob = new Blob([csv], { type: 'text/csv' });
  const url  = URL.createObjectURL(blob);
  const a    = document.createElement('a');
  a.href = url; a.download = 'nocode-ai-leads.csv'; a.click();
  URL.revokeObjectURL(url);
}
```
