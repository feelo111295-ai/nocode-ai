```javascript
// -- Onboarding --
function showOnboarding(profile) {
  const bizName = profile?.business_name || 'Your Business';
  document.getElementById('onboardingSubhead').textContent =
    'Your bots are live and working for ' + bizName + '.';

  const botsGrid = document.getElementById('onboardingBotsGrid');
  botsGrid.innerHTML = ACTIVE_BOTS.map(bot => `
    <div class="onboarding-bot-card">
      <div class="onboarding-bot-name">${bot.emoji} ${bot.name}</div>
      <div class="onboarding-bot-role">${bot.role}</div>
      <div>
        <span class="onboarding-platform-badge discord">Discord</span>
        <span class="onboarding-platform-badge telegram">Telegram</span>
      </div>
    </div>
  `).join('');

  document.getElementById('onboardingOverlay').classList.remove('hide');
}

async function dismissOnboarding() {
  const btn = document.querySelector('.onboarding-cta');
  btn.disabled = true;
  btn.textContent = 'LOADING...';

  await sb.from('profiles').update({ onboarded: true }).eq('id', currentUser.id);

  document.getElementById('onboardingOverlay').classList.add('hide');
}
```
